7.2. 消息队列

消息队列（Queue）是一种先进先出的数据结构，当两个线程需要发送某些数据结构进行通信时，可通过通过消息队列的方式发送消息，某些线程的状态会由于接收不到消息处于非运行阻塞状态，也会随着接收到其他线程或中断发送的消息恢复为运行状态进行消息处理。

7.2.1. os_taskq_post

  • 函数原型：

    int os_taskq_post(const char *name, int argc, ...);

  • 功能说明：向某个任务发送消息，消息类型为Q_MSG;

  • 参数说明：

    参数

    说明

    name

    任务名

    argc

    消息个数

  • 返回值：

    0

    消息发送成功

    非0

    消息发送失败

  • 示例：

    1void main(void)
2{
3    ... ...
4    os_taskq_post("test", 1, 0x5A); //向test任务发送消息
5    ... ...
6}

7.2.2. os_taskq_post_type

  • 函数原型：

    int os_taskq_post_type(const char *name, int type, int argc, int *argv);

  • 功能说明：向某个任务发送消息;

  • 参数说明：

    参数

    说明

    name

    任务名

    type

    消息类型：

    1. Q_MSG

    2. Q_USER

    3. Q_EVENT

    4. Q_CALLBACK

    argc

    消息个数

    argv

    消息buf

  • 返回值：

    0

    消息发送成功

    非0

    消息发送失败

  • 示例：

      1        #include "system/includes.h"
  2
  3        /**
  4        * @brief demo 1
  5        *
  6        * @return
  7        */
  8        static int _qcallback_handle()
  9        {
 10                printf("_qcallback_handle\n\n");
 11                return 0;
 12        }
 13        /**
 14        * @brief 发送callback 无参数的Q_CALLBACK的taskq demo
 15        *
 16        * @return 发送是否成功
 17        */
 18        int qcallback_msg_post_none_argv()
 19        {
 20                int argv[2];
 21                argv[0] = (int)_qcallback_handle;       // Function
 22                argv[1] = 0;                            // 参数个数
 23                return os_taskq_post_type("app_core", Q_CALLBACK, sizeof(argv) / sizeof(int), argv);
 24        }
 25        /**
 26        * @brief demo 2
 27        *
 28        * @param msg1
 29        *
 30        * @return
 31        */
 32        static int _qcallback_handle1(int msg1)
 33        {
 34                printf("_qcallback_handle1 msg1:%d\n", msg1);
 35                return 0;
 36        }
 37        /**
 38        * @brief 发送callback 1个参数的Q_CALLBACK的taskq demo
 39        *
 40        * @return 发送是否成功
 41        */
 42        int qcallback_msg_post_one_argv()
 43        {
 44                int argv[3];
 45                int msg = 1;
 46                argv[0] = (int)_qcallback_handle1;      // Function
 47                argv[1] = 1;                            // 参数个数
 48                argv[2] = msg;                          // 参数1，可以是任意类型强转成int
 49                return os_taskq_post_type("app_core", Q_CALLBACK, sizeof(argv) / sizeof(int), argv);
 50        }
 51        /**
 52        * @brief demo 3
 53        *
 54        * @param msg1
 55        * @param msg2
 56        *
 57        * @return
 58        */
 59        static int _qcallback_handle2(int msg1, int msg2)
 60        {
 61                printf("_qcallback_handle2 msg1:%d, msg2:%d\n", msg1, msg2);
 62                return 0;
 63        }
 64        /**
 65        * @brief 发送function 2个参数的Q_CALLBACK的taskq demo
 66        *
 67        * @return 发送是否成功
 68        */
 69        int qcallback_msg_post_two_argv()
 70        {
 71                int argv[4];
 72                int msg = 1;
 73                argv[0] = (int)_qcallback_handle2;      // Function
 74                argv[1] = 2;                            // 参数个数
 75                argv[2] = msg;                          // 参数1，可以是任意类型强转成int
 76                argv[3] = msg + 1;                      // 参数2，可以是任意类型强转成int
 77                return os_taskq_post_type("app_core", Q_CALLBACK, sizeof(argv) / sizeof(int), argv);
 78        }
 79        /**
 80        * @brief Demo 4
 81        *
 82        * @param msg1
 83        * @param msgArrCount
 84        * @param msgArr
 85        *
 86        * @return
 87        */
 88        static int _qcallback_handleM(int msg1, int msgArrCount, int *msgArr)
 89        {
 90                printf("_qcallback_handleM msg1:%d, msgArrCount:%d\n", msg1, msgArrCount);
 91                for (int i = 0; i < msgArrCount; i++) {
 92                        printf("_qcallback_handleM i:%d, argv:%d\n", i, msgArr[i]);
 93                }
 94                printf("_qcallback_handleM end!\n");
 95                return 100;
 96        }
 97        /**
 98        * @brief 发送function 多个（这里演示4个）参数且设置（bit8、bit9和bit10）的Q_CALLBACK的taskq demo
 99        *
100        * bit10是Function支持多参数
101        * bit8是Function的返回结果，需要在调用时把Function的返回结果的指针传入底层
102        * bit9是设置信号量，需要在调用时把信号量传入底层，在底层等待Function执行完成后会释放这个信号量
103        * （这里为了方便学习把多参数bit10、bit8和bit9做到一起，\
104        * bit9的信号量使得多参数的数组释放和bit8的回调结果在所有代码处理能在当前函数完成，\
105        * 所以推荐多参数bit10或bit8的使用和bit9一起）
106        * 注：该函数不能在Function执行的任务中调用，会死锁
107        *
108        * @return 发送是否成功
109        */
110        static int qcallback_multi_argv_num = 5;                                //参数个数5
111        int qcallback_msg_post_multi_argv_bit8_bit9()
112        {
113                int argv[4 + qcallback_multi_argv_num];
114                int *err = (int *)malloc(sizeof(int));
115                OS_SEM *sem = (OS_SEM *)malloc(sizeof(OS_SEM));
116                if (!(err && sem)) {
117                        printf("qcallback_msg_post_multi_argv_bit8_bit9 malloc error!\n");
118                        return 1;
119                }
120                argv[0] = (int)_qcallback_handleM;                  // Function
121                argv[1] = qcallback_multi_argv_num | BIT(8) | BIT(9) | BIT(10);                 // 参数个数5
122                // 以上的|BIT是使能功能bit8，bit9，bit10
123                // 用户按需设置以上argv[1]的数值
124                argv[2] = 0;                                                                            // 参数1，可以是任意类型强转成int
125                argv[3] = 1;                                                                            // 参数2
126                argv[4] = 2;                                                                            // 参数3
127                argv[5] = 3;                                                                            // 参数4
128                argv[6] = 4;                                                                            // 参数5
129                argv[7] = (int)err;                                 // BIT8，Function返回值返回到bit8
130                argv[8] = (int)sem;                                 // BIT9，Function执行完成后会释放该信号量
131                os_sem_create(sem, 0);                              // 创建信号量，初始值为0
132                int ret_err;
133                while (1) {
134                        //      执行Function的任务不能与调用qcallback_msg_post_multi_argv_bit8_bit9()的任务相同
135                        ret_err = os_taskq_post_type("btstack", Q_CALLBACK, sizeof(argv) / sizeof(int), argv);
136                        if (ret_err == OS_ERR_NONE) {
137                                printf("qcallback_msg_post_multi_argv_bit8_bit9 pend1!\n");
138                                os_sem_pend(sem, 0);    // 等待信号量，一直等待
139                                printf("qcallback_msg_post_multi_argv_bit8_bit9 pend2!\n");
140                                ret_err = *err;         // 获得Function的返回值，见_qcallback_handle_bit8_bit9的返回值是100
141                                break;
142                        }
143                        os_time_dly(2);
144                }
145                printf("qcallback_msg_post_multi_argv_bit8_bit9 free!\n");
146                free(sem);
147                free(err);
148                printf("qcallback_msg_post_multi_argv_bit8_bit9:%d!\n", ret_err);   // 这里输出100
149                return ret_err;
150        }
